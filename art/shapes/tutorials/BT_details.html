thinkBreathe:

 

Condition:

 

 

Rule:

 

%this.startStep = getCurrStep();

 

 

wait:

 

Condition:

 

if (%this.getBTActionState()==0)
 return 1;
else
 return 0;

 

Rule:

 

if ((getCurrStep() - %this.startStep) > 3)
 %this.setBTActionState(1);

 

 

playSeq:

 

Condition:

 

if (%this.getBTActionState()==1)
 return 1;
else
 return 0;

 

 

Rule:

 

%this.nextSequence = "root";
%this.loadBehaviorTree("playSequence");

 

playSequence:

 

Condition:

 

 

Rule:

 

start:

 

Condition:

 

if (%this.getBTActionState()==0)
 return 1;  
else
 return 0;

 

 

Rule:

 

%this.play(%this.nextSequence);
%this.setBTActionState(1);
%this.lastThreadPos = -1.0;

 

play:

 

Condition:

 

if ((%this.getBTActionState()==1)&&
   (%this.getThreadPos(0)>=0.0)&&
   (%this.getThreadPos(0)<1.0)&&
   (%this.getThreadPos(0)>%this.lastThreadPos)&&
   (%this.getThreadState(0)==0))
 return 1;  
else
 return 0;

 

 

Rule:

 

%this.lastThreadPos = %this.getThreadPos(0);
if (mFMod(getCurrStep(),%this.scanRate)==0)
 %this.doScan();

 

 

end:

 

Condition:

 

if ((%this.getBTActionState()==1)&&
   ((%this.getThreadState(0)==1)||(%this.getThreadPos(0)<%this.lastThreadPos)))
 return 1;
else
 return 0;

 

 

Rule:

 

%this.setBTActionState(2);

 

 

 

 

 

 

 

 

 

 

 

 

 

 

 

 

 

 

 

 

 

thinkAttack:

 

Condition:

 

return 1;

 

Rule:

 

 

startAttack:

 

Condition:

 

if (%this.getBTActionState()==0)
 return 1;  
else
 return 0;

 

 

Rule:

 

%diff = VectorLen(VectorSub(%this.getTarget().getPosition(),%this.getPosition()));
if ((%diff < %this.attackMaxRange))
{
 %this.play(%this.attackSequence);
 %this.lastThreadPos = -1.0;
 %this.setBTActionState(1);
}

 

 

strike:

 

Condition:

 

if (%this.getBTActionState()==1)
 return 1;
else
 return 0;

 

Rule:

 

if (%this.getThreadPos(0)>%this.attackStart)
{
 if ((%this.getTarget().getKinematic())&&
     ( strcmp(%this.getTarget().getBTName(),"thinkRecover") )&&//if NOT thinkRecover
     ( strcmp(%this.getTarget().getBTName(),"playSequence") ))//and NOT playSequence
 {
   echo("doing attack, target BTName " @ %this.getTarget().getBTName() );
   %this.doAttack();
 }
 %this.setBTActionState(2);
}

 

 

endAttack:

 

Condition:

 

if (%this.getBTActionState()==2)
 return 1;  
else
 return 0;

 

Rule:

 

if ((%this.getThreadPos(0) == 1.0) ||
   (%this.getThreadPos(0) < %this.lastThreadPos))
{
 %this.setBTActionState(3);
}
%this.lastThreadPos = %this.getThreadPos(0);

getToPosition:

 

Condition:

 

%diff = VectorSub(%this.getPosition(), %this.getMoveGoal());
if (VectorLen(%diff) > %this.getMoveThreshold())
 return 1;
else
 return 0;

 

 

Rule:

 

 

 

moveToPos:

 

Condition:

 

%diff = VectorSub(%this.getPosition(), %this.getMoveGoal());
if ((%this.hasNavPath()==false)&&(VectorLen(%diff) > %this.getMoveThreshold()))
 return 1;
else
 return 0;

 

Rule:

 

%this.setMoveTarget(%this.getMoveGoal());
%this.loadBehaviorTree("moveToPosition");

 

 

 

 

 

followPath:

 

Condition:

 

if (%this.hasNavPath()==true)
 return 1;
else
 return 0;

 

 

Rule:

 

%this.setNavPathTo(%this.getMoveGoal());
%this.setNavPathNode(0);

 

 

 

travel:

 

Condition:

 

if (%this.getNavPathNode()  < %this.getNavPathCount()-1)
 return 1;
else
 return 0;

 

 

Rule:

 

%this.setNavPathNode(%this.getNavPathNode() + 1);
%this.setMoveTarget(%this.getNavPathNodePos(%this.getNavPathNode()));
%this.loadBehaviorTree("moveToPosition");

 

 

 

 

 

 

 

 

 

 

 

 

 

 

 

moveToPosition:

 

Condition:

 

%diff = VectorSub(%this.getPosition(), %this.getMoveTarget());
if (VectorLen(%diff) > %this.getMoveThreshold())
 return 1;
else
 return 0;

 

 

Rule:

 

 

 

start:

 

Condition:

 

if (%this.getBTActionState()==0)
 return 1;  
else
 return 0;

 

 

Rule:

 

if (%this.hasNavPath())
 %this.moveToPosition( %this.getMoveTarget(),%this.getMoveSequence() );
else
 %this.moveToPosition( %this.getMoveTarget(),%this.getMoveSequence() );//steerToPosition?
%this.scanFunction = "PatrolScan";
%this.setBTActionState(1);

 

 

 

 

 

 

travel:

 

Condition:

 

%diff = VectorSub(%this.getPosition(), %this.getMoveTarget());
if ((%this.getBTActionState()==1)&&
   (VectorLen(%diff) > %this.getMoveThreshold()) )
 return 1;  
else
 return 0;

 

 

Rule:

 

if (mFMod(getCurrStep(),%this.scanRate)==0)
 %this.doScan();

 

 

 

end:

 

Condition:

 

%diff = VectorSub(%this.getPosition(), %this.getMoveTarget());
if ((%this.getBTActionState()==1)&&
   (VectorLen(%diff)  <= %this.getMoveThreshold()) )
 return 1;  
else
 return 0;

 

 

Rule:

 

%this.setBTActionState(2);

startMoving:

 

Condition:

 

if ((%this.hasNavPath()==true)&&
   (%this.getBTActionState()==0))
 return 1;
else
 return 0;

 

 

Rule:

 

%this.setMoveSequence("run");
%this.setNavPathTo(%this.getMoveGoal());
%this.setBTActionState(1);

 

 

 

 

 

 

travel:

 

Condition:

 

if (%this.getBTActionState()==1)
 return 1;
else
 return 0;

 

 

Rule:

 

if (VectorLen(%this.getVehicleVel()) < 3.5)
 %this.setMoveSequence("walk");
else
 %this.setMoveSequence("run");
%this.updateToVehicle();

Condition:

 return 1;
else
 return 0;

 

Rule:

wanderOut:

 

Condition:

 return 1;
else
 return 0;

Condition:

else
